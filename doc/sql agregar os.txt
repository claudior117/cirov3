Actualizar segun S, P, C items de items_os
UPDATE items_os
SET aplica = 'S'
WHERE SUBSTRING(codigo, 1, 2) = '01';

UPDATE items_os
SET aplica = 'C'
WHERE SUBSTRING(codigo, 1, 2) = '02';

UPDATE items_os
SET aplica = 'P'
WHERE SUBSTRING(codigo, 1, 2) <> '01' and SUBSTRING(codigo, 1, 2) <> '02';
